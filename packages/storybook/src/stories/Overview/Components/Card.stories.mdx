    },
    {

      defaultValue: "'default'",
      required: false,
      description: 'Choose the logo in the corner'
    },
    {

      defaultValue: '1',
      required: false,

    },
    {
      name: 'customLogo',
      type: 'React.ReactNode',
      defaultValue: '',
      required: false,
      description:
        'The logo that wil be displayed in the corner instead of the defaults logos'
    },
    {
      name: 'logoSize',
      type: "'medium' | 'small'",
      defaultValue: "'medium'",
      required: false,
      description: 'Defined the size of the logo in the corner'
    },
    {
      name: 'separator',
      type: "React.ReactNode | 'none'",
      defaultValue: '',
      required: false,
      description:
